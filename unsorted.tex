\section*{Approximating binomial coefficients}

\begin{lemma}[Stirling's approximation, \cite{Robbins1955}]\label{lem:stirling approx}
  \[n! &\in \sqrt{2\pi n} \left(\frac{n}{e}\right)^n \cdot \left[e^{\frac{1}{12n + 1}}, e^\frac{1}{12n}\right] & n > 0\]
\end{lemma}

\begin{lemma}\label{lem:binom bounds}
  \[
  \left(\frac{n}{k}\right)^k &\le \binom{n}{k} \le \frac{n^k}{k!} \le \left(\frac{en}{k}\right)^k & n \ge k > 0
  \]
\end{lemma}
\begin{proof}
  \[
  \prod_{i=0}^{k-1} \frac{n}{k}
  &\le \prod_{i=0}^{k-1} \frac{n - i}{k - i} \le \prod_{i=0}^{k-1} \frac{n}{k - i}
  \]
  together with \cref{lem:stirling approx}.
\end{proof}

\begin{lemma}\label{lem:binom frac}
  Let $0 < k \le n_1 \le n_2$.  Then
  \[
  \frac{\binom{n_1}{k}}{\binom{n_2}{k}} \le \left(\frac{n_1}{n_2}\right)^k.
  \]
  Equivalently, the probability of drawing $k$ red balls in a row from an urn without replacement is lower than with replacement.
\end{lemma}
\begin{proof}
  Write out the products
  \[
  \frac{\prod_{i=0}^{k-1} \frac{n_1 - i}{k - i}}{\prod_{i=0}^{k-1} \frac{n_2 - i}{k - i}}.
  \]
\end{proof}

\section*{Combining multiplicative approximations}

\begin{lemma}[Bernoulli's inequality]\label{lem:bernoulli}
  \[
    (1 + x)^r &\le 1 + rx & x > -1, r \in [0, 1] \\
    (1 + x)^r &\ge 1 + rx & x > -1, r \in \R \setminus (0, 1) \\
    (1 + x)^r &\ge 1 + rx & x \ge -2, r \in \N \\
    (1 + x)^r &\ge 1 + rx & x \in \R, \text{$r \in \N$ even} \\
    \]
\end{lemma}

\begin{lemma}
  Suppose $x_i \in [1 - \epsilon_i, 1 + \eta_i]$ for each $i \in [n]$, where $\epsilon_i \in [0, 1]$ and $\eta_i \ge -1$.
  Then
  \[
  \prod_i x_i
  &\in \left[1 - \sum_i \epsilon_i, e^{\sum_i \eta_i}\right],
  \]
  which is contained in $\left[1 - \sum_i \epsilon_i, 1 + (e - 1) \sum_i \eta_i\right]$ if $\sum_i \eta_i \in [0, 1]$.
\end{lemma}
\begin{proof}
  The lower bound follows by induction:
  \[
  \prod_{i=1}^{n+1} x_i &\ge \left(1 - \sum_{i=1}^n \epsilon_i\right) (1 - \epsilon_{n+1}) \\
  &= 1 - \epsilon_{n+1} - \sum_{i=1}^n \epsilon_i + \epsilon_{n+1} \sum_{i=1}^n \epsilon_i \\
  &\ge 1 - \sum_{i=1}^{n+1} \epsilon_i
  \]
  For the upper bound:
  \[
  \prod_i x_i
  &\le \exp \sum_i \log (1 + \eta_i) \\
  &\le \exp\left[n \log \left(\frac1n \sum_i (1 + \eta_i)\right)\right] &\text{by concavity} \\
  &= \left(1 + \frac1n \sum_i \eta_i\right)^n \\
  &\le e^{\sum_i \eta_i}
  \]
\end{proof}

\section*{Other}

Let $X_1, \dots, X_n$ be independent random variables $X_i \sim \Ber\left(\frac{1 - \delta_i}{2}\right)$.
Then
\[\bigoplus_i X_i \sim \Ber\left(\frac{1 - \prod_i \delta_i}{2}\right).\]

\begin{proof}
  Let $Y_i = (-1)^{X_i}$; then $Y_1, \dots, Y_n$ are independent Rademacher random variables so
\[
1 - 2\Pr\left[\bigoplus_i X_i = 1\right] = \E\left[\prod_i Y_i\right] = \prod_i \E[Y_i] = \prod_i \delta_i.
\]
\end{proof}
